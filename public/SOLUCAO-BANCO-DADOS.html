📋 SCRIPT SQL PARA CRIAR BANCO

Execute este script no phpMyAdmin para criar o banco de dados:

-- Criar banco de dados CREATE DATABASE IF NOT EXISTS sistema_agendamento; USE sistema_agendamento; -- Tabela de usuários CREATE TABLE IF NOT EXISTS usuarios ( id INT AUTO_INCREMENT PRIMARY KEY, nome VARCHAR(100) NOT NULL, email VARCHAR(100) UNIQUE NOT NULL, senha VARCHAR(255) NOT NULL, tipo_usuario ENUM('aluno', 'professor', 'admin') DEFAULT 'aluno', criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Tabela de cursos CREATE TABLE IF NOT EXISTS cursos ( id INT AUTO_INCREMENT PRIMARY KEY, nome VARCHAR(100) NOT NULL, descricao TEXT, carga_horaria INT DEFAULT 40, criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Tabela de certificados CREATE TABLE IF NOT EXISTS certificados ( id INT AUTO_INCREMENT PRIMARY KEY, aluno_id INT NOT NULL, curso_id INT NOT NULL, codigo_verificacao VARCHAR(50) UNIQUE NOT NULL, data_emissao DATE NOT NULL, data_conclusao_curso DATE NOT NULL, status ENUM('emitido', 'validado', 'revogado') DEFAULT 'emitido', criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (aluno_id) REFERENCES usuarios(id), FOREIGN KEY (curso_id) REFERENCES cursos(id) ); ❌ PROBLEMAS COMUNS

  1. MySQL não está rodando: Inicie o MySQL no XAMPP
  2. Banco não existe: Execute o script SQL acima
  3. Credenciais incorretas: Verifique o arquivo db.php
  4. Tabelas vazias: Insira dados de teste
  5. Permissões: Verifique permissões do MySQL
